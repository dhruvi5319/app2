**API Services:**

`src/services/weatherApi.ts` — `fetchWeatherData(location: LocationResult): Promise<WeatherData>`:
- URL: `https://api.open-meteo.com/v1/forecast?latitude={lat}&longitude={lon}&timezone=auto&current=temperature_2m,apparent_temperature,weather_code,is_day,wind_speed_10m,wind_direction_10m,relative_humidity_2m&hourly=temperature_2m,weather_code,is_day,precipitation_probability&daily=weather_code,temperature_2m_max,temperature_2m_min,precipitation_probability_max,sunrise,sunset,uv_index_max,wind_speed_10m_max,wind_direction_10m_dominant&forecast_days=7&wind_speed_unit=kmh`
- `timezone=auto` is **non-negotiable** — always included
- `transformForecastResponse(raw, location)`:
  - All temps: `Math.round()`
  - `is_day: 1 → true, 0 → false`
  - Hourly: find current hour index in `hourly.time` array → slice 24 forward (if not enough hours, pad from start of next day)
  - `null` precipitation → 0
  - `null` uvIndexMax → 0
  - `fetchedAt: Date.now()`
  - `precipitationProbability` for current conditions sourced from `daily.precipitation_probability_max[0]` (NOT from current block)
- Export: `fetchWeatherData`, `transformForecastResponse`

`src/services/geocodingApi.ts` — `searchCity(query: string): Promise<GeocodingResult[]>`:
- URL: `https://geocoding-api.open-meteo.com/v1/search?name={query}&count=5&language=en&format=json`
- Returns `[]` when `response.results === undefined` (no error thrown)
- Transforms each result to `GeocodingResult` shape

`src/services/nominatimApi.ts` — `reverseGeocode(lat: number, lon: number): Promise<string>`:
- URL: `https://nominatim.openstreetmap.org/reverse?lat={lat}&lon={lon}&format=json&zoom=10`
- Required header: `User-Agent: SimpleWeatherApp/1.0 (https://weather-app.vercel.app)`
- Returns city name: `address.city ?? address.town ?? address.village ?? display_name.split(",")[0]`

**Custom Hooks:**

`src/hooks/useWeatherData.ts` — TanStack Query hook:
```typescript
import { useQuery } from "@tanstack/react-query";
import { LocationResult } from "../types/weather";
import { queryKeys } from "../constants/queryKeys";
import { fetchWeatherData } from "../services/weatherApi";

export function useWeatherData(location: LocationResult | null) {
  return useQuery({
    queryKey: location ? queryKeys.weather(location.latitude, location.longitude) : ["weather-disabled"],
    queryFn: () => fetchWeatherData(location!),
    enabled: location !== null,
  });
}
```

`src/hooks/useGeocodingSearch.ts` — TanStack Query hook with 300ms debounce:
- Uses `useState` for debounced query; `useQuery` enabled only when `debouncedQuery.length >= 2`
- `queryKey: queryKeys.geocoding(debouncedQuery)`
- `queryFn: () => searchCity(debouncedQuery)`
- Returns `{ results: GeocodingResult[], isLoading, isDebouncing }`

`src/hooks/useGeolocation.ts`:
- `requestLocation()` calls `navigator.geolocation.getCurrentPosition`
- On `PERMISSION_DENIED (code 1)`: silently reset state (no error to caller)
- On `POSITION_UNAVAILABLE (code 2)`: provide error string to caller for toast
- Returns `{ requestLocation, isLocating, error }`

`src/hooks/useUnitPreference.ts`:
- Reads from `readUnitPreference()` on init
- `toggle()` flips celsius↔fahrenheit and calls `writeUnitPreference()`
- Returns `{ unit, toggle }`

`src/hooks/useRecentSearches.ts`:
- Reads from `readRecentSearches()` on init
- `addSearch(location: LocationResult)` calls `writeRecentSearch()` and refreshes state
- Returns `{ searches: RecentSearch[], addSearch }`

`src/hooks/useFreshnessTimer.ts`:
- Accepts `fetchedAt: number | undefined`
- Ticks every 60 seconds via `setInterval` to force re-render
- Returns `formattedFreshness: string` from `formatFreshness(fetchedAt)`

`src/hooks/useReverseGeocode.ts`:
- One-shot hook; calls `reverseGeocode(lat, lon)` then `searchCity(cityName)`
- Returns first `GeocodingResult` as `LocationResult` shape

```tsx
import { useState, useRef } from "react";
import { LocationResult, WeatherData } from "./types/weather";
import { AppLayout } from "./components/layout/AppLayout";
import { Footer } from "./components/layout/Footer";
import { SearchBar } from "./components/search/SearchBar";
import { HeroSection } from "./components/weather/HeroSection";
import { HourlyStrip } from "./components/weather/HourlyStrip";
import { SkeletonHourly } from "./components/feedback/SkeletonHourly";
import { DailyForecastList } from "./components/weather/DailyForecastList";
import { SkeletonDaily } from "./components/feedback/SkeletonDaily";
import { TemperatureTrendChart } from "./components/weather/TemperatureTrendChart";
import { ChartErrorBoundary } from "./error-boundaries/ChartErrorBoundary";
import { AppErrorBoundary } from "./error-boundaries/AppErrorBoundary";
import { useWeatherData } from "./hooks/useWeatherData";
import { useUnitPreference } from "./hooks/useUnitPreference";

// Inner component has access to hooks
function WeatherApp() {
  const [activeLocation, setActiveLocation] = useState<LocationResult | null>(null);
  const { unit, toggle } = useUnitPreference();
  const { data: weatherData, isLoading, isError } = useWeatherData(activeLocation);
  const announcerRef = useRef<HTMLDivElement>(null);

  // Announce location changes to screen readers
  const handleLocationSelect = (location: LocationResult) => {
    setActiveLocation(location);
    if (announcerRef.current) {
      announcerRef.current.textContent = `Loading weather for ${location.name}`;
    }
  };

  const handleUnitToggle = () => {
    toggle();
  };

  return (
    <AppLayout>
      {/* Global aria-live announcer */}
      <div
        role="status"
        aria-live="polite"
        aria-atomic="true"
        className="sr-only"
        ref={announcerRef}
      />

      <SearchBar onLocationSelect={handleLocationSelect} />

      {activeLocation ? (
        <>
          <HeroSection
            location={activeLocation}
            unit={unit}
            onUnitToggle={handleUnitToggle}
          />

          {/* Hourly strip */}
          {isLoading && <SkeletonHourly />}
          {weatherData && (
            <HourlyStrip
              hourly={weatherData.hourly}
              timezone={activeLocation.timezone}
              unit={unit}
            />
          )}

          {/* Daily forecast + chart */}
          {isLoading && <SkeletonDaily />}
          {weatherData && (
            <>
              <div className="w-full lg:flex lg:gap-4">
                <div className="lg:flex-1">
                  <DailyForecastList
                    daily={weatherData.daily}
                    timezone={activeLocation.timezone}
                    unit={unit}
                  />
                </div>
                <div className="lg:flex-1 mt-4 lg:mt-0">
                  <ChartErrorBoundary
                    daily={weatherData.daily}
                    timezone={activeLocation.timezone}
                    unit={unit}
                  >
                    <TemperatureTrendChart
                      daily={weatherData.daily}
                      timezone={activeLocation.timezone}
                      unit={unit}
                      locationName={activeLocation.name}
                    />
                  </ChartErrorBoundary>
                </div>
              </div>
            </>
          )}
        </>
      ) : (
        <div className="flex flex-col items-center justify-center py-16 text-white/60">
          <p className="text-lg">Search for a city to see weather</p>
        </div>
      )}

      <Footer />
    </AppLayout>
  );
}

export default function App() {
  return (
    <AppErrorBoundary>
      <WeatherApp />
    </AppErrorBoundary>
  );
}
```
